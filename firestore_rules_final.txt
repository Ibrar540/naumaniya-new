rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 🔐 USER PROFILE DATA
    match /users/{userId} {

      // Allow account owner to read/update/delete their data
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;

      // Allow creating a new account for authenticated users
      allow create: if request.auth != null;

      // Allow 'get' (like showing account info) to allowed users
      // FIXED: Removed duplicate 'allow get' since 'allow read' already covers this
    }

    // 📥 ACCESS REQUESTS (Subcollection: users/{userId}/requests/{requestId})
    match /users/{userId}/requests/{requestId} {

      // Allow any logged-in user to send a request
      allow create: if request.auth != null;

      // Allow only the target account (userId) to read/update their incoming requests
      allow read, update: if request.auth != null && request.auth.uid == userId;

      // Disallow deletion of requests (optional safety)
      allow delete: if false;
    }

    // 📁 OTHER DATA (students, teachers, budget, etc.)
    match /{collection}/{document=**} {

      // Allow read/write for authenticated users
      // You can make this more restrictive based on your needs
      allow read, write: if request.auth != null;
    }
  }
}
