rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 🔐 USER PROFILE DATA
    match /users/{userId} {
      allow create: if request.auth != null;
      
      // Only the account owner can view or edit their own account
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
    }

    // 📥 ACCESS REQUESTS
    match /users/{userId}/requests/{requestId} {
      // Any logged-in user can send a request
      allow create: if request.auth != null;

      // Only the receiver (account owner) can view/respond
      allow read, update: if request.auth != null && request.auth.uid == userId;

      // No deletion of requests
      allow delete: if false;
    }

    // 📁 USER-OWNED DATA (students, teachers, budget, etc.)
    match /users/{userId}/{subCollection}/{docId} {
      allow read, write: if request.auth != null && (
        request.auth.uid == userId || (
          get(/databases/$(database)/documents/users/$(userId)).data.allowedUsers != null &&
          request.auth.uid in get(/databases/$(database)/documents/users/$(userId)).data.allowedUsers
        )
      );
    }

    // 📊 GENERAL DATA COLLECTIONS (if you store data at root level)
    // This is for backward compatibility if your app stores data directly in collections
    match /students/{docId} {
      allow read, write: if request.auth != null;
    }
    
    match /teachers/{docId} {
      allow read, write: if request.auth != null;
    }
    
    match /income/{docId} {
      allow read, write: if request.auth != null;
    }
    
    match /expenditure/{docId} {
      allow read, write: if request.auth != null;
    }

    // 🛑 Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
