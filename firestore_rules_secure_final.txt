rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 🔐 USER PROFILE DATA
    match /users/{userId} {
      // Allow creation if the user is creating their own document
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Only the account owner can view or edit their own account
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
    }

    // 📥 ACCESS REQUESTS
    match /users/{userId}/requests/{requestId} {
      // Any logged-in user can send a request
      allow create: if request.auth != null;

      // Only the receiver (account owner) can view/respond
      allow read, update: if request.auth != null && request.auth.uid == userId;

      // No deletion of requests (audit trail)
      allow delete: if false;
    }

    // 🔓 OPEN ACCESS FOR AUTHENTICATED USERS
    // This is a temporary solution to get the app working.
    // For production, you should implement more granular security rules.
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
