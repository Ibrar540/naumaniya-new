rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 🔐 USER PROFILE DATA
    match /users/{userId} {
      // Allow creation if the user is creating their own document
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // Only the account owner can view or edit their own account
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
    }

    // 📥 ACCESS REQUESTS
    match /users/{userId}/requests/{requestId} {
      // Any logged-in user can send a request
      allow create: if request.auth != null;

      // Only the receiver (account owner) can view/respond
      allow read, update: if request.auth != null && request.auth.uid == userId;

      // No deletion of requests (audit trail)
      allow delete: if false;
    }

    // 📁 USER-OWNED DATA (students, teachers, budget, etc.)
    // This is the recommended structure: /users/{userId}/students/{docId}
    match /users/{userId}/{subCollection}/{docId} {
      // 🔒 STRICT OWNERSHIP VALIDATION
      allow create: if request.auth != null && 
        request.auth.uid == userId;  // Only owner can create in their own subcollection
      
      allow read, write: if request.auth != null && (
        request.auth.uid == userId || (
          get(/databases/$(database)/documents/users/$(userId)).data.allowedUsers != null &&
          request.auth.uid in get(/databases/$(database)/documents/users/$(userId)).data.allowedUsers
        )
      );
    }

    // 🚫 ROOT-LEVEL COLLECTIONS - ENHANCED SECURITY
    // If your app uses root-level collections, enforce strict ownership
    match /students/{docId} {
      // 🔒 CREATE: Must include userId matching authenticated user
      allow create: if request.auth != null && 
        request.resource.data.userId == request.auth.uid;
      
      // 🔒 READ/WRITE: Only document owner or allowed users
      allow read, update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        // Check if user is in allowedUsers of the document owner
        (get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers != null &&
         request.auth.uid in get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers)
      );
    }
    
    match /teachers/{docId} {
      allow create: if request.auth != null && 
        request.resource.data.userId == request.auth.uid;
      
      allow read, update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        (get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers != null &&
         request.auth.uid in get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers)
      );
    }
    
    match /income/{docId} {
      allow create: if request.auth != null && 
        request.resource.data.userId == request.auth.uid;
      
      allow read, update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        (get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers != null &&
         request.auth.uid in get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers)
      );
    }
    
    match /expenditure/{docId} {
      allow create: if request.auth != null && 
        request.resource.data.userId == request.auth.uid;
      
      allow read, update, delete: if request.auth != null && (
        resource.data.userId == request.auth.uid ||
        (get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers != null &&
         request.auth.uid in get(/databases/$(database)/documents/users/$(resource.data.userId)).data.allowedUsers)
      );
    }

    // 🛑 Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
