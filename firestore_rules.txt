rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // USER PROFILE DATA
    match /users/{userId} {
      
      // 🔓 Allow the user to read or write their own account
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null;

      // 🔒 Allow only users listed in 'allowedUsers' to read this user's data
      allow get: if request.auth != null &&
                  (request.auth.uid == userId ||
                   request.auth.uid in resource.data.allowedUsers);
    }

    // MANAGE REQUESTS (stored as subcollection under users/{userId}/requests/{requestId})
    match /users/{userId}/requests/{requestId} {
      // Anyone logged in can send a request to any user
      allow create: if request.auth != null;

      // Only the receiver (userId) can read or update the request
      allow read, update: if request.auth != null && request.auth.uid == userId;
    }

    // GENERAL DATA COLLECTIONS (students, teachers, income, expenditure, etc.)
    match /{collection}/{document=**} {
      // Allow read/write for authenticated users
      // You can make this more restrictive based on your needs
      allow read, write: if request.auth != null;
    }
  }
}
